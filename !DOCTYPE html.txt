    <script>
        // 1. 舒心地新增行程細項（並自動在景點間補上交通填寫欄位）
        function addNewSpotBlock() {
            const container = document.getElementById('timelineContainer');
            
            // 如果前一個區塊還沒有交通欄位，先幫它補上，維持時間軸的完整流暢
            const lastBlock = container.querySelector('.timeline-block:last-child');
            if (lastBlock && !lastBlock.querySelector('.transit-item')) {
                const transitDiv = document.createElement('div');
                transitDiv.className = 'transit-item';
                transitDiv.innerHTML = `
                    <span>🚗</span>
                    <input type="text" class="transit-input" placeholder="輸入往下一站的交通方式與時間...">
                `;
                lastBlock.appendChild(transitDiv);
            }

            // 建立全新的行程項目細項
            const newBlock = document.createElement('div');
            newBlock.className = 'timeline-block';
            newBlock.innerHTML = `
                <div class="timeline-item">
                    <div class="item-row">
                        <input type="time" class="time-input" value="18:00" oninput="updateRouteSummary()">
                        <input type="text" class="spot-input" placeholder="輸入新景點/地點名稱..." oninput="updateRouteSummary()">
                        <button class="btn-delete" onclick="deleteSpot(this)">🗑️</button>
                    </div>
                </div>
            `;
            
            container.appendChild(newBlock);
            updateRouteSummary(); // 即時更新底部的路線串聯
        }

        // 2. 隨時拿掉不想要的細項行程
        function deleteSpot(button) {
            const block = button.closest('.timeline-block');
            block.remove();
            updateRouteSummary(); // 移除後，路線摘要自動重新計算
        }

        // 3. 自動抓取左側景點，在下方即時生成「A → B → C」的路線摘要
        function updateRouteSummary() {
            const spotInputs = document.querySelectorAll('.spot-input');
            const routeSummaryList = document.getElementById('routeSummaryList');
            
            routeSummaryList.innerHTML = ''; // 清空舊資料
            
            let filledSpots = [];
            spotInputs.forEach((input) => {
                if(input.value.trim() !== "") {
                    filledSpots.push(input.value.trim());
                }
            });

            // 串聯並印出帶有優雅箭頭的路線
            filledSpots.forEach((spot, index) => {
                const spotSpan = document.createElement('span');
                spotSpan.style.fontWeight = '500';
                spotSpan.innerText = spot;
                routeSummaryList.appendChild(spotSpan);

                if (index < filledSpots.length - 1) {
                    const arrowSpan = document.createElement('span');
                    arrowSpan.className = 'route-arrow';
                    arrowSpan.innerText = '  →  ';
                    routeSummaryList.appendChild(arrowSpan);
                }
            });

            if(filledSpots.length === 0) {
                routeSummaryList.innerHTML = '<span style="color: var(--text-muted); font-size: 0.85rem;">暫無規劃行程</span>';
            }
        }

        // 初始化載入
        window.onload = function() {
            updateRouteSummary();
        }
    </script>
